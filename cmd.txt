#Assemble the Bootloader
nasm -f bin boot.asm -o boot.bin

#compile kernel
gcc -m16 -ffreestanding -fno-PIE -nostartfiles -nostdlib -c kernel.c -o kernel.o

#Link kernel
ld -m elf_i386 -T link.ld --oformat=binary kernel.o -o kernel.bin

#Create a Test Disk Image
dd if=/dev/zero of=DA-Os.img bs=512 count=2880
dd if=boot.bin of=DA-Os.img conv=notrunc
dd if=kernel.bin of=DA-Os.img bs=512 seek=1 conv=notrunc

#Test Using an Emulator
qemu-system-i386 -fda DA-Os.img
